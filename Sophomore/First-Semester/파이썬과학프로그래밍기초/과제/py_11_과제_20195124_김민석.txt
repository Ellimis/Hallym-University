#영어단어와 한글해석이 빈칸으로 구분되도록 입력 받아 단어장에 저장하고 출력하세요.

dic = {}

while True:
    s = input("영어단어와 한글해석을 빈칸으로 구분해 입력하세요 : ")
    if s == '': break
    #strip 으로 양쪽 공백을 제거
    s = s.strip()

    #문자열.split(' ')을 통해 첫번째 인덱스의 영어와 맨 끝에 있는 인덱스의 한국어를 가지고 옴
    eng = s.split(' ')[0]
    kor = s.split(' ')[-1]
    dic.update({eng:kor})

print("--------- 저장된 단어장은 -------------------------------")
print(dic)
