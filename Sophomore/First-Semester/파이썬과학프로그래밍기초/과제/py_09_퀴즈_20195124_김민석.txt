lambda 또는 comprehension을 이용하여 다음과 같이 동작하는 프로그램을 작성합니다.

>>>

print("입력 받은 수까지의 n의 배수를 만듭니다.")
n = int(input("n? "))
by = int(input("어디까지 할까요? "))

li = [i for i in range(1, by+1) if i%n == 0]
print(li)

plus = int(input("위 리스트의 모든 원소에 더할 값? "))

print(list(map(lambda x : x+plus, li)))
