#include <stdio.h>
#include <string.h>

void ex_01() {
	/*  예비문제 1. 정수형 변수에 사용자가 입력한 정수를 저장하고 포인터를 이용하여 출력하시오.
		예비문제 2. 문자형 변수에 사용자가 입력한 문자를 저장하고 포인터를 이용하여 출력하시오.
	*/
	int in, *p;
	char ch, *q;

	printf("\n정수 입력 ");
	scanf_s("%d", &in);
	p = &in; //주소 치환
	printf("\n입력된 정수는 %d", *p);

	getchar();

	printf("\n문자 입력 ");
	scanf_s("%c", &ch);
	q = &ch;
	printf("\n입력된 문자는 %c", *q);

	getchar();
}

void ex_03() {
	//예비문제 4. 정수배열 변수에 사용자가 입력한 정수 다섯 개를 저장하고 포인터를 이용하여 출력하시오.
	int in[5], *p, i;

	for (i = 0; i < sizeof(in) / sizeof(int); i++) {
		printf("\n정수 입력 ");
		scanf_s("%d", in+i);
	}
	
	p = in;
	
	for (i = 0; i < sizeof(in) / sizeof(int); i++) {
		printf("%d\t", *(p+i));
	}
}
