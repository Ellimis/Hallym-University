#include <stdio.h>

int last_num(int a, int b) {
	return (a <= b) ? b % a : a % b;
}

int first_num(int a, int b) {
	return (a <= b) ? b / a : a / b;
}

void ex_02() {
	/*
	Q2) 2개의 정수를 입력받아 큰 수를 작은 수로 나눈 몫과 나머지를 각각 출력하는
	프로그램을 작성하시오. ( 몫을 반환하는 함수, 나머지를 반환하는 함수 2개 )
	*/
	int a, b, q, r;
	printf("두개의 정수를 입력 : \n");
	scanf_s("%d%*c%d", &a, &b); //%*c : 두개 정수 사이에 띄어쓰기 점 등을 넣어도 인식
	getchar();

	q = first_num(a, b);
	r = last_num(a, b);

	printf("\n두 정수의 몫은 %d, 나머지는 %d 입니다.", q, r);
}

void main(void) {
	ex_02();
	getchar();
}
