#include <stdio.h>

void printgugu(int *p, int size, int cnt) {
	int j, i;

	for (i = cnt; i <= cnt; i++) {
		printf("%d단 출력\n", cnt);
		for (j = 1; j <= size; j++) {
			printf("%d * %d = %d\n", cnt, j, (cnt*j));
		}
	}
}

	사용자가 출력을 원하는 단 입력 받음
	printgugu() 는 배열, 사이즈, 입력 받은 수를 매개변수로 받아 이중 포문으로 원하는 구구단 출력하는 함수
	*/
	int cnt, gugudan[9];

	printf("\n구구단에서 출력을 원하는 단을 입력 : ");
