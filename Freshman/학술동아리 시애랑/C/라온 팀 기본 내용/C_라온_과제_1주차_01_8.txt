#include <stdio.h>

int myfactorial(int n) {
	if (n == 1) return 2;
	else return 2 * myfactorial(n - 1);
}

void main()
{
	/*
	문제 8
		2의 N승을 구하는 함수를 재귀적으로 구현해 보자. 그리고 그에 따른 적절한 main()함수도 구현해보자.

		실행의 예
		정수 입력: 8
		2의 8승은 256
	*/
	int n;

	printf("정수 입력 : ");
	scanf_s("%d", &n);
	getchar();

	printf("2의 %d승은 %d", n, myfactorial(n));
	getchar();
}
