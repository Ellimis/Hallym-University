#include <stdio.h>

void ex_06() {
	/*
	문제 6
		프로그램 사용자로부터 초(second)를 입력 받은 후에, 
		이를 [시, 분, 초]의 형태로 출력하는 프로그램을 작성해 보자.
		실행의 예
		초(second) 입력: 3615
		[h:1, m:0, s:15]
	*/
	int h = 0, m = 0, s = 0;

	printf("초(second) 입력 : ");
	scanf_s("%d", &s);

	h = s / 3600;
	m = s % 3600 / 60;
	s = s % 3600 % 60;
	printf("[h : %d], [m : %d], [s : %d]", h, m, s);
	getchar();
}

void main(void)
{
	ex_06();
	getchar();
}
