#include <stdio.h>

void ex_01() {
	/*
	문제 1
		문자 4개를 저장하는 배열을 선언하고 사용자로부터 문자를 입력 받아 저장하고 출력하시오. 
		아래 참고
		문자입력: c
		문자입력: g
		문자입력: v
		문자입력: s
		입력된 문자 = cgvs
	*/
	char str[4];

	for (int i = 0; i < sizeof(str) / sizeof(str[0]); i++) {
		printf("\n문자입력 : ");
		scanf_s("%c", &str[i]);
		getchar();
	}
	printf("\n입력된 문자 = ");
	for (int i = 0; i < sizeof(str) / sizeof(str[0]); i++) {
		printf("%c", str[i]);
	}
}

void main(void)
{
	ex_01();
	getchar();
}
