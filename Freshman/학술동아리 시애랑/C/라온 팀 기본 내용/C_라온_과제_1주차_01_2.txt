#include <stdio.h>

void gugudan(int x, int y) {
	int max, min, i, j;

	if (x >= y) {
		max = x;
		min = y;
	}
	else {
		max = y;
		min = x;
	}

	for (i = min; i <= max; i++) {
		printf("--------- %d단 출력 ---------\n", i);
		for (j = 1; j <= 9; j++) {
			printf("\t%d * %d = %d\n", i, j, i*j);
		}
	}
}
