#include <stdio.h>

void ex_06() {
	/*
	문제 6
		문자열을 입력 받아 거꾸로 저장하고 출력하는 작업을 반복하는 프로그램 작성하시오.
		단, <Enter키>만을 입력 받으면 반복을 종료하시오. 아래 참고.

		문자열입력 : apple
		역순=elppa
		문자열입력 : cat
		역순=tac
		문자열입력 : <Enter키>
	*/
	char ch[50], hc[50];
	int i, len;

	while (1) {
		i = 0;
		len = 0;

		printf("\n문자열 입력 : ");
		gets(ch);
		if (ch[0] == '\0') break;

		while (ch[i]) {
			len++;
			i++;
		}

		i = 0;
		for (int j = len - 1; j >= 0; j--, i++) {
			hc[i] = ch[j];
		}

		hc[i] = '\0';
		printf("역순 : %s", hc);
	}
}

void main(void)
{
	ex_06();
	getchar();
}
