#include <stdio.h>

void ex_07() {
	/*
	문제 7
		프로그램 사용자로부터 숫자 n을 입력 받는다. 그리고 나서 다음 공식이 성립하는 
		k의 최댓값을 계산해서 출력하는 프로그램을 작성해 보자.
		2^k≤n

		실행의 예
		상수 n 입력: 256
		공식을 만족하는 k의 최댓값은 8
	*/
	int n, cmp = 1, k = 0;

	printf("\n상수 n 입력 : ");
	scanf_s("%d", &n);

	while (cmp <= n) {
		cmp *= 2;
		k++;
	}
	printf("공식을 만족하는 k의 최댓값은 %d", k - 1);
	getchar();
}

void main(void)
{
	ex_07();
	getchar();
}
