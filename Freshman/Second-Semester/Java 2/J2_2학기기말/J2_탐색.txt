import java.util.*;

public class J2_test {	
	public static void main(String[] args) {
		int key = 50;
		List<Integer> list = new ArrayList<>();
		for(int i = 0; i < 100; i++) {
			list.add(i*2);
		}
		int index = Collections.binarySearch(list, key);
		System.out.println("탐색의 반환 값 = " + index);
	}
}
