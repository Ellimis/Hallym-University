import java.util.*;

public class J2_test {	
	public static void main(String[] args) {
		int time = 10;
		Queue<Integer> queue = new LinkedList<>();
		for(int i = time; i >= 0; i--) {
			queue.add(i);
		}
		while(!queue.isEmpty()) {
			System.out.println(queue.remove() + " ");
		}
	}
}
