void ex_13() {
	/*13.	4개 문자열을 입력 받아 2차원 문자열 배열에 저장하고 각각 문자열의 길이를 출력하시오. 
	문자열입력 : cgvs
	문자열입력 : cgv
	문자열입력 : cg
	문자열입력 : c
	cgvs=4
	cgv=3
	cg=2
	c=1
	*/
	char ch[4][20];
	int i, j;

	for (i = 0; i < sizeof(ch) / sizeof(ch[0]); i++) {
		printf("\n문자열 입력 : ");
		gets(ch[i]);
	}

	for (i = 0; i < sizeof(ch) / sizeof(ch[0]); i++) {

		j = 0;		
		while(ch[i][j]) {
		    j++;
		}
		printf("\n%s = %d", ch[i], strlen(ch[i]));
		printf("\n%s = %d", ch[i], j);
	}
}

void ex_14() {
	/*14.다섯 사람의 이름을 표준입력으로 입력 받아 저장하고 저장된
		이름 중 원하는 사람의 이름이 있으면 다음과 같이 출력하는 프로그램을 작성하시오.
		저장될 이름 입력 : ann
		저장될 이름 입력 : benjamin
		저장될 이름 입력 : jhon
		저장될 이름 입력 : kate
		저장될 이름 입력 : tom

		찾을 이름 입력 : Benjamin
		benjamin 우리 반 입니다.

		찾을 이름 입력 : amy
		amy 우리 반이 아닙니다.

		찾을 이름 입력 : <Enter>
	*/
	char ch[5][10];
	char fin[10];
	int cnt = 0, j, i;

	for (i = 0; i < sizeof(ch) / sizeof(ch[0]); i++) {
		printf("\n저장될 이름 입력 : ");
		gets(ch[i]);
		j = 0;

		while (ch[i][j]) {
			ch[i][j] = tolower(ch[i][j]);
			j++;
		}
		//ch = tolower(ch); 문자 ch 에 한 문자를 소문자로 바꿈
	}

	while (1) {
		printf("\n찾을 이름 입력 : ");
		gets(fin);

		if (!fin[0]) break;
		i = 0;

		while (fin[i]) {
			fin[i] = tolower(fin[i]);
			i++;
		}
		cnt = 0;

		for (i = 0; i < sizeof(ch) / sizeof(ch[0]); i++) {
			if (!strcmp(fin, ch[i])) cnt++;
		}

		if (cnt) {
			printf("\n%s 우리 반 입니다.", fin);
		}
		else {
			printf("\n%s 우리 반이 아닙니다.", fin);
		}
	}
}

void ex_15() {
	/*15.2차원 문자열 배열 사용하여 문자열 2개를 
		입력 받아 같은지 다른지 비교하는 프로그램을 작성하시오.
		문자열입력 : apple
		문자열입력 : apples
		apple, apples 다름.

		문자열입력 : apple
		문자열입력 : apple
		apple, apple 같음.

		문자열입력 : <Enter키>
	*/
	char ch[2][15];
	int i;

	while (1) {
		i = 0;
		printf("\n문자열 입력 : ");
		gets(ch[0]);
		if (ch[0][0] == '\0') break;

		printf("\n문자열 입력 : ");
		gets(ch[1]);
		if (ch[1][0] == '\0') break;

		if (!(strcmp(ch[0], ch[1]))) printf("%s, %s 같음", ch[0], ch[1]);
		else printf("%s, %s 다름", ch[0], ch[1]);
		i++;
	}
}

	ex_13();

}
