#include <stdio.h>


void ex_02() {
	//2.	지불할 금액을 정수로 입력 받아 화폐단위가 각각 몇 개씩 필요한지 출력하는 프로그램을 작성하시오.
	int money, a, b, c, d;
	printf("금액을 입력해주세요! \n");
	scanf_s("%d", &money);
	a = money / 50000;
	b = money % 50000 / 10000;
	c = money % 10000 / 5000;
	d = money % 5000 / 1000;
	printf("50000원권 %d개, 10000원권 %d개, 5000원권 %d개, 1000원권 %d개 \n", a, b, c, d);
	getchar();
}

void ex_04() {
	//4.	세 수를 입력 받아 가장 큰 수를 구하는 프로그램을 조건 연산자를 이용하여 작성하시오.
	int a, b, c;
	printf("세 수를 입력하세요.\n");
	scanf_s("%d%d%d", &a, &b, &c);
	(a >= b) ?
		(a >= c) ? printf("a가 가장 크다.") : printf("c가 가장 크다.")
		:
		(b >= c) ? printf("b가 가장 크다.") : printf("c가 가장 크다.");
	getchar();
}

void main(void) {
	
	ex_02();
	ex_04();
	getchar();
}
