void ex_05() {
	int a, count=0;

	printf("정수 입력");
	scanf_s("%d", &a);

	for (; a>0; a>>=1)
		if (a & 1) {
			count++;
		}
	
	printf("1이 나온 횟수는 %d번이다.", count);
	getchar();
}

void ex_06() {
	char ch;
	int count = 0;
	printf("문자 입력");
	ch = _getche();

	if ((!isalpha(ch))) return;
	
	for (ch = tolower(ch); ch <= 'z'; ch++) {
		printf("%c%c", ch, (!(count++ % 5)? '\n' : ' '));
	}
	getchar();
}
