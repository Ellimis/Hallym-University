#include <stdio.h>
#include<stdlib.h>
#include<math.h>

void ex_14() {
	int i, j;

	for (i = 0; i <= 7; i++)
	{
		for (j = -7; j <= 7; j++)
		{
			if (abs(j) <= i)
				printf("%d", abs(j));
			else
				printf(" ");
		}
		printf("\n");
	}
	getchar();
}

void ex_15() {
	int i, cnt, dis = 100;
	char ch, temp;

	printf("반복 횟수 입력 : \n");
	scanf_s("%d", &cnt);

	for (i = 1; i <= cnt; i++) {

		printf("\n문자입력 : ");
		scanf_s(" %c", &ch);

		if (islower(ch))
		{
			if (abs('m' - ch) < dis)
			{
				temp = ch;
				dis = abs('m' - ch);
			}
		}
		else if (isupper(ch))
		{
			if (abs('M' - ch) < dis)
			{
				temp = ch;
				dis = abs('M' - ch);
			}
		}
		else
		{
			printf("=>알파벳이 아닙니다. 다시 입력합니다.");
			cnt++;
		}
	}
	if (islower(ch)) printf("m과 가장 가까운 문자는 %c", temp);
	else if (isupper(ch)) printf("M과 가장 가까운 문자는 %c", temp);

	getchar();
}



void main(void) {
	ex_14();
	ex_15();
	getchar();
}
